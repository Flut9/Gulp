Gulp - сборщик проектов (автоматизирует рутинные вещи)
Установка:
    1.Set-ExecutionPolicy RemoteSigned (в powershell от админа)
    2.npm install --global gulp-cli (устанавливает глобально)
    3.npm init 
    4.npm install gulp --save-dev (устанавливает для проекта)
    5.Создаем gulpfile.js 

npm cache clean --force //чистит менеджет пакетов
npm i npm -g //устанавливаем пакетный менеджер
npm i gulp-sass --save-dev 


Сценарии:
    let project_folder = 'dist'; //папка, куда будет все складываться
    let source_folder = 'src'; //папка с исходниками

    let path = {
        build: {
            html: project_folder + '/', //путь для выгрузки html файлов
            css: project_folder + '/css/',
            js: project_folder + '/js/',
            img: project_folder + '/img/',
        },
        src: {
            html: source_folder + '/',
            css: source_folder + '/scss/style.scss', //все scss файлы будут складываться в style.scss
            js: source_folder + '/js/script.js', //все js файлы будут складываться в script.js
            img: source_folder + '/img/**/*.{jpg, png, svg, gif, ico, webp}', //будут просматриваться только картинки с расширениями jpg, png, ...
        },
        watch: {
            html: source_folder + '/**/*.html',
            css: source_folder + '/scss/**/*.scss', 
            js: source_folder + '/js/script.js', 
            img: source_folder + '/img/**/*.{jpg, png, svg, gif, ico, webp}', 
        }
    }
